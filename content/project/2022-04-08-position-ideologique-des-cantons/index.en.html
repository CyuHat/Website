library(tidyverse)
library(FactoMineR)
library(factoextra)
load("votation.Rda")
v_pca <-
  votation %>% 
  group_by(canton,id_objet) %>% 
  filter(date==last(date)) %>% 
  select(-date) %>% 
  ungroup() %>% 
  spread(id_objet,oui_en,drop = TRUE) 
pca <- 
  v_pca %>%
  column_to_rownames("canton") %>% 
  PCA(graph = FALSE)
